string SzyfrCezara(string ciagDoZaszyfrowania_Bez_Tablicy, int przesuniecie)
{

char ciagDoZaszyfrowania[dlugoscCiaguDoZaszyfrowania + 1];

int b = 0;


for (int x = 0; x < dlugoscCiaguDoZaszyfrowania; x++)
{
for (int a = 0; a < 32; a++)
{

if (ciagDoZaszyfrowania[x] == alfabet[a])
{
b = a + przesuniecie;
if (b > 31)
{
b = b - 32;
}

}

}
