char *tr(char *str)
{
static char buff[256];
char cp[]="\245\206\251\210\344\242\230\276\253\244\217\250\235\343\340\227\275\215ąćęłńóśżźĄĆĘŁŃÓŚŻŹ";
if(strlen(str)>=sizeof(buff)) return str;
char *bf=buff;
while(*str)
{
char *pos=strchr(cp+18,*str);
*(bf++)=pos?*(pos-18):*str;
++str;
}
*bf=0;
return buff;
}




int main()
{
unsigned long long i, j;
long double aktualnaLiczba, sumaLiczb, srednia;
long double liczby[9999];
i = 0;
j = 0;
aktualnaLiczba = 0;
sumaLiczb = 0;
srednia = 0;

cout << tr("Wprowadź liczby do wyliczenia średniej oddzielone spacją: ");
string liczbyDoPosortowania;
getline(cin, liczbyDoPosortowania);

string liczby_W_Stringu[9999];
istringstream ss(liczbyDoPosortowania);
string word;

while (ss >> word)
{
liczby_W_Stringu[j] = word;
j++;
}

for (i = 0; i < j; i++)
{
aktualnaLiczba = stof(liczby_W_Stringu[i]);
liczby[i] = aktualnaLiczba;
}

for (i = 0; i < j; i++)
{
sumaLiczb = liczby[i] + sumaLiczb;
}

srednia = sumaLiczb / j;

cout << tr("\nŚrednia wprowadzonych liczb: ") << srednia;
return 0;
}
