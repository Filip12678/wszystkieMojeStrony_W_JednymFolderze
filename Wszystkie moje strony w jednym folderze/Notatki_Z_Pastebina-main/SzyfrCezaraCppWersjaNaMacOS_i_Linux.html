std::string ReplaceAll(std::string str, const std::string& from, const std::string& to) {
size_t start_pos = 0;
while((start_pos = str.find(from, start_pos)) != std::string::npos) {
str.replace(start_pos, from.length(), to);
start_pos += to.length(); // Handles case where 'to' is a substring of 'from'
}
return str;
}



using namespace std;



string SzyfrCezara(string ciagDoZaszyfrowania_Bez_Tablicy, int przesuniecie)
{
//Definiowanie alfabetu
string alfabet1 = "A\244BC\217DE\250FGHIJKL\235MN\343O\340PRS\227TUWYZ\215\275";
int dlugoscAlfabetu = alfabet1.length();
char alfabet[dlugoscAlfabetu + 1];
strcpy(alfabet, alfabet1.c_str());


int dlugoscCiaguDoZaszyfrowania = ciagDoZaszyfrowania_Bez_Tablicy.length();
char ciagDoZaszyfrowania[dlugoscCiaguDoZaszyfrowania + 1];
strcpy(ciagDoZaszyfrowania, ciagDoZaszyfrowania_Bez_Tablicy.c_str());


int b = 0;


for (int x = 0; x < dlugoscCiaguDoZaszyfrowania; x++)
{
for (int a = 0; a < 32; a++)
{

if (ciagDoZaszyfrowania[x] == alfabet[a])
{
b = a + przesuniecie;
if (b > 31)
{
b = b - 32;
}
ciagDoZaszyfrowania[x] = alfabet[b];
break;
}
}
}
return ciagDoZaszyfrowania;
}
