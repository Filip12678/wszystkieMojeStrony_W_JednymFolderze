#include <iostream>
#include <bits/stdc++.h>

using namespace std;


bool comparator(string a,string b)
{
return a<b;
}


int main()
{
unsigned long long i, j;
i = 0;
j = 0;

cout << "Wprowad\253 wyrazy do posortowania w kolejno\230ci alfabetycznej oddzlielone spacj\245: ";
string wyrazyDoPosortowania;
getline(cin, wyrazyDoPosortowania);

string wyrazy[9999];
istringstream ss(wyrazyDoPosortowania);
string word;

while (ss >> word)
{
wyrazy[j] = word;
j++;
}

sort(wyrazy,wyrazy+j,comparator);

cout << "\nPosortowane wyrazy:\n\n";

for (i = 0; j > i ; i++)
{
cout << wyrazy[i] << "\n";
}

getchar();getchar();

return 0;
}
